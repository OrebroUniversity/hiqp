132 template<typename Derived>
133 Derived dls
134 (
135  const Eigen::MatrixBase<Derived>& a,
136  double eta = 0.01
137 )
138 {
139  unsigned int r = a.rows();
140  unsigned int c = a.cols();
141 
142  if (r > c)
143  return a;
144 
145  if (r == c)
146  return pinv(a);
147 
148  Derived a_ext(r+c, c);
149  a_ext.block(0, 0, r, c) = a;
150  a_ext.block(r, 0, c, c) = eta * Eigen::MatrixXd::Identity(c, c);
151  Derived b = Eigen::MatrixXd::Zero(r+c, r);
152  b.block(0, 0, r, r) = Eigen::MatrixXd::Identity(r, r);
153 
154  return pinv(a_ext) * b;
155 }
