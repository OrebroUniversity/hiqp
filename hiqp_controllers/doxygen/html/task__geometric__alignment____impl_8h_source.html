58 template<typename PrimitiveA, typename PrimitiveB>
59 int TaskGeometricAlignment<PrimitiveA, PrimitiveB>::init
60 (
61  const HiQPTimePoint& sampling_time,
62  const std::vector<std::string>& parameters,
63  const KDL::Tree& kdl_tree,
64  unsigned int num_controls
65 )
66 {
67 
68  if (parameters.size() != 4)
69  return -1;
70 
71  std::stringstream ss(parameters.at(2));
72  std::vector<std::string> args(
73  std::istream_iterator<std::string>{ss},
74  std::istream_iterator<std::string>{});
75 
76  if (args.size() != 3)
77  return -2;
78 
79  e_.resize(1);
80  J_.resize(1, num_controls);
81  e_dot_star_.resize(1);
82  performance_measures_.resize(1);
83 
84  primitive_a_ = geometric_primitive_map_->getGeometricPrimitive<PrimitiveA>(args.at(0));
85  primitive_b_ = geometric_primitive_map_->getGeometricPrimitive<PrimitiveB>(args.at(2));
86 
87  geometric_primitive_map_->addDependencyToPrimitive(args.at(0), this->getTaskName());
88  geometric_primitive_map_->addDependencyToPrimitive(args.at(2), this->getTaskName());
89 
90  int sign = 0;
91 
92  if (args.at(1).compare("<") == 0 ||
93  args.at(1).compare("<=") == 0)
94  {
95  sign = -1;
96  }
97  else if (args.at(1).compare("=") == 0 ||
98  args.at(1).compare("==") == 0)
99  {
100  sign = 0;
101  }
102  else if (args.at(1).compare(">") == 0 ||
103  args.at(1).compare(">=") == 0)
104  {
105  sign = 1;
106  }
107  else
108  {
109  return -3;
110  }
111 
112  delta_ = std::stod( parameters.at(3) );
113 
114  task_types_.insert(task_types_.begin(), 1, sign);
115 
116  return 0;
117 }

225 template<typename PrimitiveA, typename PrimitiveB>
226 int TaskGeometricAlignment<PrimitiveA, PrimitiveB>::alignVectors
227 (
228  const KDL::Vector& v1,
229  const KDL::Vector v2
230 )
231 {
232  double d = KDL::dot(v1, v2);
233 
234  e_(0) = d - std::cos(delta_);
235 
236  //std::cout << "e = " << e_(0) << "\n";
237 
238  KDL::Vector v = v1 * v2; // v = v1 x v2
239 
240  for (int q_nr = 0; q_nr < jacobian_a_.columns(); ++q_nr)
241  {
242  KDL::Vector Ja = jacobian_a_.getColumn(q_nr).rot;
243  KDL::Vector Jb = jacobian_b_.getColumn(q_nr).rot;
244 
245  J_(0, q_nr) = KDL::dot( v, (Ja - Jb) );
246  }
247 
248  return 0;
249 }
